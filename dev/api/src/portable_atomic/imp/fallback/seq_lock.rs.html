5include!("seq_lock_common.rs");

7use core::sync::atomic::AtomicU64;
8
9pub(super) type State = u64;
10
11/// A simple stamped lock.
12pub(super) struct SeqLock {
13    /// The current state of the lock.
14    ///
15    /// All bits except the least significant one hold the current stamp. When locked, the state
16    /// equals 1 and doesn't contain a valid stamp.
17    state: AtomicU64,
18}
19
20impl SeqLock {
21    #[inline]
22    pub(super) const fn new() -> Self {
23        Self { state: AtomicU64::new(0) }
24    }
25
26    /// If not locked, returns the current stamp.
27    ///
28    /// This method should be called before optimistic reads.
29    #[inline]
30    pub(super) fn optimistic_read(&self) -> Option<State> {
31        // The acquire load from `state` synchronizes with the release stores in
32        // `SeqLockWriteGuard::drop` and `SeqLockWriteGuard::abort`.
33        let state = self.state.load(Ordering::Acquire);
34        if state == LOCKED { None } else { Some(state) }
35    }
36
37    /// Returns `true` if the current stamp is equal to `stamp`.
38    ///
39    /// This method should be called after optimistic reads to check whether they are valid. The
40    /// argument `stamp` should correspond to the one returned by method `optimistic_read`.
41    #[inline]
42    pub(super) fn validate_read(&self, stamp: State) -> bool {
43        // Thanks to the fence, if we're noticing any modification to the data at the critical
44        // section of `stamp`, then the critical section's write of 1 to state
45        // should be visible.
46        crate::fence(Ordering::Acquire);
47        self.state.load(Ordering::Relaxed) == stamp
48    }
49}
50
51impl SeqLockWriteGuard<'_> {
52    #[inline]
53    fn next_stamp(&self) -> State {
54        self.state.wrapping_add(2)
55    }
56}
