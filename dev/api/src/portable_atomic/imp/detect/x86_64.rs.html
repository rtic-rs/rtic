4Run-time CPU feature detection on x86_64 by using CPUID.

9#![cfg_attr(portable_atomic_sanitize_thread, allow(dead_code))]
10
11// Miri doesn't support inline assembly used in __cpuid: https://github.com/rust-lang/miri/issues/932
12// SGX doesn't support CPUID: https://github.com/rust-lang/rust/blob/1.92.0/library/std_detect/src/detect/os/x86.rs#L30-L33
13#[cfg(any(target_env = "sgx", miri))]
14compile_error!("internal error: this module is not supported on this environment");
15
16include!("common.rs");
17
18#[cfg(not(portable_atomic_no_asm))]
19use core::arch::asm;
20use core::arch::x86_64::CpuidResult;
21
22// Workaround for https://github.com/rust-lang/rust/issues/101346
23// It is not clear if our use cases are affected, but we implement this just in case.
24//
25// Refs:
26// - https://www.felixcloutier.com/x86/cpuid
27// - https://en.wikipedia.org/wiki/CPUID
28// - https://github.com/rust-lang/stdarch/blob/a0c30f3e3c75adcd6ee7efc94014ebcead61c507/crates/core_arch/src/x86/cpuid.rs
29#[cfg(not(target_env = "sgx"))]
30fn __cpuid(leaf: u32) -> CpuidResult {
31    let eax;
32    let mut ebx;
33    let ecx;
34    let edx;
35    // SAFETY: Calling `__cpuid` is safe on all x86_64 CPUs except for SGX,
36    // which doesn't support `cpuid`.
37    // https://github.com/rust-lang/stdarch/blob/a0c30f3e3c75adcd6ee7efc94014ebcead61c507/crates/core_arch/src/x86/cpuid.rs#L102-L109
38    unsafe {
39        asm!(
40            "mov r8, rbx", // save rbx which is reserved by LLVM
41            "cpuid",
42            "xchg r8, rbx", // restore rbx
43            out("r8") ebx,
44            inout("eax") leaf => eax,
45            inout("ecx") 0 => ecx,
46            out("edx") edx,
47            options(nostack, preserves_flags),
48        );
49    }
50    CpuidResult { eax, ebx, ecx, edx }
51}
52
53#[cold]
54fn _detect(info: &mut CpuInfo) {
55    let CpuidResult { ecx: proc_info_ecx, .. } = __cpuid(1);
56
57    // https://github.com/rust-lang/rust/blob/1.92.0/library/std_detect/src/detect/os/x86.rs#L104
58    if test(proc_info_ecx, 13) {
59        info.set(CpuInfoFlag::cmpxchg16b);
60    }
61
62    // We only use VMOVDQA when SSE is enabled. See _atomic_load_vmovdqa() in atomic128/x86_64.rs for more.
63    #[cfg(target_feature = "sse")]
64    {
65        use core::arch::x86_64::_xgetbv;
66
67        // https://github.com/rust-lang/rust/blob/1.92.0/library/std_detect/src/detect/os/x86.rs#L166-L236
68        let cpu_xsave = test(proc_info_ecx, 26);
69        if cpu_xsave {
70            let cpu_osxsave = test(proc_info_ecx, 27);
71            if cpu_osxsave {
72                // SAFETY: Calling `_xgetbv` is safe because the CPU has `xsave` support
73                // and OS has set `osxsave`.
74                let xcr0 = unsafe { _xgetbv(0) };
75                let os_avx_support = xcr0 & 6 == 6;
76                if os_avx_support && test(proc_info_ecx, 28) {
77                    info.set(CpuInfoFlag::avx);
78                }
79            }
80        }
81    }
82}

91#[cfg(test)]
92mod tests {
93    use super::*;
94
95    #[test]
96    #[cfg_attr(portable_atomic_test_detect_false, ignore = "detection disabled")]
97    fn test_cpuid() {
98        // The recent Rosetta 2 unofficially implements AVX support.
99        // (The OS reports it as unsupported, likely due to poor performance.)
100        #[cfg(target_vendor = "apple")]
101        assert_eq!(
102            std::is_x86_feature_detected!("avx"),
103            detect().avx() || cfg!(target_feature = "avx")
104        );
105        #[cfg(not(target_vendor = "apple"))]
106        assert_eq!(std::is_x86_feature_detected!("avx"), detect().avx());
107        assert_eq!(std::is_x86_feature_detected!("cmpxchg16b"), detect().cmpxchg16b());
108    }
