1use crate::syntax::ast::App;
2use crate::{analyze::Analysis, codegen::util};
3use proc_macro2::TokenStream as TokenStream2;
4use quote::quote;

6/// Generates `local` variables and local resource proxies
7///
8/// I.e. the `static` variables and theirs proxies.
9pub fn codegen(app: &App, _analysis: &Analysis) -> TokenStream2 {
10    let mut mod_app = vec![];
11
12    // All local resources declared in the `#[local]' struct
13    for (name, res) in &app.local_resources {
14        let cfgs = &res.cfgs;
15        let ty = &res.ty;
16        let mangled_name = util::static_local_resource_ident(name);
17
18        let attrs = &res.attrs;
19
20        // late resources in `util::link_section_uninit`
21        // unless user specifies custom link section
22        let section = if attrs.iter().any(|attr| {
23            let is_link_section = attr.path().is_ident("link_section");
24            let is_unsafe = attr.path().is_ident("unsafe");
25            let is_embedded_link_section = match attr.parse_args() {
26                Ok(syn::Expr::Assign(assign)) => match &*assign.left {
27                    syn::Expr::Path(path) => path.path.is_ident("link_section"),
28                    _ => false,
29                },
30                _ => false,
31            };
32
33            is_link_section || (is_unsafe && is_embedded_link_section)
34        }) {
35            None
36        } else {
37            Some(util::link_section_uninit())
38        };
39
40        // For future use
41        // let doc = format!(" RTIC internal: {}:{}", file!(), line!());
42        mod_app.push(quote!(
43            #[allow(non_camel_case_types)]
44            #[allow(non_upper_case_globals)]
45            // #[doc = #doc]
46            #[doc(hidden)]
47            #(#attrs)*
48            #(#cfgs)*
49            #section
50            static #mangled_name: rtic::RacyCell<core::mem::MaybeUninit<#ty>> = rtic::RacyCell::new(core::mem::MaybeUninit::uninit());
51        ));
52    }
53
54    // All declared `local = [NAME: TY = EXPR]` local resources
55    for (task_name, resource_name, task_local) in app.declared_local_resources() {
56        let cfgs = &task_local.cfgs;
57        let ty = &task_local.ty;
58        let expr = &task_local.expr;
59        let attrs = &task_local.attrs;
60
61        let mangled_name = util::declared_static_local_resource_ident(resource_name, task_name);
62
63        // For future use
64        // let doc = format!(" RTIC internal: {}:{}", file!(), line!());
65        mod_app.push(quote!(
66            #[allow(non_camel_case_types)]
67            #[allow(non_upper_case_globals)]
68            // #[doc = #doc]
69            #[doc(hidden)]
70            #(#attrs)*
71            #(#cfgs)*
72            static #mangled_name: rtic::RacyCell<#ty> = rtic::RacyCell::new(#expr);
73        ));
74    }
75
76    quote!(#(#mod_app)*)
77}
