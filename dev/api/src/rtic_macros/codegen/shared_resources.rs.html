8/// Generates `static` variables and shared resource proxies
9pub fn codegen(app: &App, analysis: &Analysis) -> TokenStream2 {
10    let mut mod_app = vec![];
11    let mut mod_resources = vec![];
12
13    for (name, res) in &app.shared_resources {
14        let cfgs = &res.cfgs;
15        let ty = &res.ty;
16        let mangled_name = &util::static_shared_resource_ident(name);
17
18        let attrs = &res.attrs;
19
20        // late resources in `util::link_section_uninit`
21        // unless user specifies custom link section
22        let section = if attrs.iter().any(|attr| {
23            let is_link_section = attr.path().is_ident("link_section");
24            let is_unsafe = attr.path().is_ident("unsafe");
25            let is_embedded_link_section = match attr.parse_args() {
26                Ok(syn::Expr::Assign(assign)) => match &*assign.left {
27                    syn::Expr::Path(path) => path.path.is_ident("link_section"),
28                    _ => false,
29                },
30                _ => false,
31            };
32
33            is_link_section || (is_unsafe && is_embedded_link_section)
34        }) {
35            None
36        } else {
37            Some(util::link_section_uninit())
38        };
39
40        // For future use
41        // let doc = format!(" RTIC internal: {}:{}", file!(), line!());
42        mod_app.push(quote!(
43            #[allow(non_camel_case_types)]
44            #[allow(non_upper_case_globals)]
45            // #[doc = #doc]
46            #[doc(hidden)]
47            #(#attrs)*
48            #(#cfgs)*
49            #section
50            static #mangled_name: rtic::RacyCell<core::mem::MaybeUninit<#ty>> = rtic::RacyCell::new(core::mem::MaybeUninit::uninit());
51        ));
52
53        // For future use
54        // let doc = format!(" RTIC internal: {}:{}", file!(), line!());
55
56        let shared_name = util::need_to_lock_ident(name);
57
58        if !res.properties.lock_free {
59            mod_resources.push(quote!(
60                // #[doc = #doc]
61                #[doc(hidden)]
62                #[allow(non_camel_case_types)]
63                #(#cfgs)*
64                pub struct #shared_name<'a> {
65                    __rtic_internal_p: ::core::marker::PhantomData<(&'a (), *const u8)>,
66                }
67
68                // Opt out from `Send`.
69                // `#shared_name` is trivially `Sync` because there are no `&self` methods.
70                // See https://doc.rust-lang.org/std/sync/struct.Exclusive.html .
71                #(#cfgs)*
72                unsafe impl<'a> Sync for #shared_name<'a> {}
73
74                #(#cfgs)*
75                impl<'a> #shared_name<'a> {
76                    #[inline(always)]
77                    pub unsafe fn new() -> Self {
78                        #shared_name { __rtic_internal_p: ::core::marker::PhantomData }
79                    }
80                }
81            ));
82
83            let ptr = quote!(
84                #(#cfgs)*
85                #mangled_name.get_mut() as *mut _
86            );
87
88            let ceiling = match analysis.ownerships.get(name) {
89                Some(Ownership::Owned { priority } | Ownership::CoOwned { priority }) => *priority,
90                Some(Ownership::Contended { ceiling }) => *ceiling,
91                None => 0,
92            };
93
94            // For future use
95            // let doc = format!(" RTIC internal ({} resource): {}:{}", doc, file!(), line!());
96
97            mod_app.push(impl_mutex(
98                app,
99                analysis,
100                cfgs,
101                true,
102                &shared_name,
103                &quote!(#ty),
104                ceiling,
105                &ptr,
106            ));
107        }
108    }
109
110    let mod_resources = if mod_resources.is_empty() {
111        quote!()
112    } else {
113        quote!(mod shared_resources {
114            #(#mod_resources)*
115        })
116    };
117
118    quote!(
119        #(#mod_app)*
120
121        #mod_resources
122    )
123}
