44pub struct Instance<T, const N: u8> {
45    ptr: core::ptr::NonNull<T>,
46}
47impl<T, const N: u8> core::ops::Deref for Instance<T, N> {
48    type Target = T;
49    #[inline]
50    fn deref(&self) -> &Self::Target {
51        unsafe { self.ptr.as_ref() }
52    }
53}
54impl<T, const N: u8> Instance<T, N> {
